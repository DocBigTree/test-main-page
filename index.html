Quick Start 
pip install optverse-solver

from optverse import Solver

# Create solver instance
solver = Solver()

# Define your optimization problem
problem = solver.create_problem()
problem.add_variable('x', lb=0, ub=10)
problem.add_variable('y', lb=0, ub=10)
problem.set_objective('maximize', '3*x + 2*y')
problem.add_constraint('x + y <= 8')

# Solve
solution = solver.solve(problem)
print(f"Optimal value: {solution.objective_value}")
